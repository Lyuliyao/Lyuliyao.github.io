Memory Address 

          
            #include <iostream>
            using namespace std;
            int main() {
                int i = 42;
                cout << i << endl;            // Output the value of i
                cout << &i << endl;        // Output the memory address of i

                int* p = &i;               // Pointer p holds the address of i
                cout << p << endl;              // Output the address stored in p
                cout << *p << endl;             // Output the value at the address stored in p

                *p = 1;
                cout << i << endl;            // Output the modified value of i
                cout << *p << endl;             // Output the modified value via pointer p
                return 0;
            }

Null pointer 

            
            #include <iostream>
            using namespace std;
            int main(){
                int i = 0;
                int* pi = &i; // pi is a pointer that holds the address of i
                int j = 1;
                int* pj = &j; // pj is a pointer that holds the address of j
                int* p = nullptr; // p is a null pointer
                p = pi; // p now points to i
                cout << &i << endl; // Output the address of i
                cout << p << endl; // Output the address stored in p
                cout << *p << endl; // Output the value of i via pointer p
                p = pj; // p now points to j
                cout << &j << endl; // Output the address of j
                cout << p << endl; // Output the address stored in p
                cout << *p << endl; // Output the value of j via pointer p
                return 0;
            }

Pointer to Pointer 

            
            #include <iostream>
            using namespace std;
            int main(){
                int i = 42;
                int* p = &i; // p is a pointer to i
                int** pp = &p; // pp is a pointer to pointer p
                cout << i << endl; // Output the value of i
                cout << *p << endl; // Output the value of i via pointer p
                cout << **pp << endl; // Output the value of i via pointer to pointer pp
                return 0;
            }

* and & 

            
            #include <iostream>
            using namespace std;
            int main(){
                int i = 42;
                int& ri = i; // ri is a reference to i // int& means ri is a reference to an int
                int* p = &i; // p is a pointer to i; int* means p is a pointer to an int
                cout << *p << endl; // Output the value of i via pointer p; *p is dereferencing pointer p
                cout << &i << endl; // Output the address of i; &i is the address of i
                cout << &ri << endl; // Output the address of i via reference ri; &ri is the address of i
                return 0;
            }

Pointer and const 

            
            #include <iostream>
            using namespace std;
            int main(){
                double d1 = 1.1;
                double d2 = 2.2;

                double* _p_d = &d1;           // pointer to double
                const double* _pcd = &d1;     // pointer to const double
                double* const cp_d = &d1;     // const pointer to double
                const double* const cpcd = &d1; // const pointer to const

                _p_d = &d2;        // OK: change pointer to another address
                _pcd = &d2;       // OK: change pointer to another address
                //cp_d = &d2;      // Error: cannot change const pointer to another address
                //cpcd = &d2;     // Error: cannot change const pointer to another address

                *_p_d = 3.3;         // OK: change value at the address
                // *_pcd = 4.4;      // Error: cannot change value at
                *cp_d = 5.5;       // OK: change value at the address
                // *cpcd = 6.6;    // Error: cannot change value at the address
                return 0;
            }

swap again 

            
            #include <iostream>
            using namespace std;
            void swap(int* p1, int* p2){
                int temp = *p1;
                *p1 = *p2;
                *p2 = temp;
            }
            int main(){
                int a = 1;
                int b = 2;
                cout << "Before swap: a = " << a << ", b = " << b << endl;
                swap(&a, &b);
                cout << "After swap: a = " << a << ", b = " << b << endl;
                return 0;
            }
