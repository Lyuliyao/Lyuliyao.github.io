Classes

class

  • By default, all members of a class are private
  • private members can only be accessed by member functions of the class

Example 

          
            #include <iostream>
            using namespace std;
            class A{
              public:
                A() : i(0) {}
                void print_i() const { cout << i << ' '; f(); }
                void mutate_i() { ++i; f(); }
                void mutate_other_i(A& other) const;
              private:
                int i;
                void f() const {}
            };
            void A::mutate_other_i(A& other) const { ++other.i; other.f(); }
            int main() {
              A a0, a1;
              // a0.i; // Builderror.
              // a0.f(); // Builderror.
              a0.print_i();
              a0.mutate_i();
              a0.print_i();
              a1.print_i();
              a0.mutate_other_i(a1);
              a1.print_i();
              return 0;
            }

Build error

  • It is a build error to access a private member of a struct or class outside of the definitions of the class’s member functions and constructors

            #include <iostream>

            void f(int x, int y = 7, int z = 42) {

            }

              cout << x << ' ' << y << ' ' << z << endl;
