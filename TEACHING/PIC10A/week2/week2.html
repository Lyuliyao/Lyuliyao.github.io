Section 6: Strings

Constucting std::string 


                #include <string>
                #include <iostream>
                #include <vector>
                using namespace std;

                int main() {
                    string s1 = "Hello world";
                    string s2("hello");
                    string s3(8, 'a');
                    string s0;

                    cout << s1 << endl;  
                    cout << s2 << endl; 
                    cout << s3 << endl; 
                    cout << s0 << endl; 
                    return 0;
                }

Remark

  • std::string is a class in the C++ standard library
  • s is an instance of std::string , but not a variable of type std::string
  • A class is a souped-up data type, with member functions (built-in functions)

member functions 


                #include <string>
                #include <iostream>
                using namespace std;
                int main() {
                    string s = "Hello world";
                    cout << s.size() << endl;  //
                    cout << s.length() << endl; // same as size()
                    cout << s.empty() << endl;  //
                    cout << s.at(0) << endl;    // character at index 0
                    cout << s.substr(0, 5) << endl; // substring from index 0, length 5
                    return 0;
                }

Indexing 


                #include <string>
                #include <iostream>
                using namespace std;
                int main() {
                    string s = "Hello world";
                    
                    cout << s[0] << endl; // character at index 0
                    cout << s[1] << endl; // character at index 1
                    cout << s[10] << endl; // character at index 10
                }
