Section 9: Functions

Create a function 

          
#include <iostream>
using namespace std;
int myFunction() {
  cout << "I just got executed!";
  return 0;
}

int main() {
  myFunction(); // call the function
  cout << endl;
  return 0;
}

Call a Function 

          
#include <iostream>
using namespace std;
void myFunction() {
  cout << "I just got executed!" << endl;
}

int main() {
  myFunction(); // call the function
  cout << endl;
  myFunction(); // call the function
  cout << endl;
  myFunction(); // call the function
  cout << endl;
  return 0;
}

Declaration and Definition of functions

          
#include <iostream>
using namespace std;
void myFunction(); // function declaration


int main() {
  myFunction(); // call the function
  cout << endl;
  myFunction(); // call the function
  cout << endl;
  myFunction(); // call the function
  cout << endl;
  return 0;
}

void myFunction() { // function definition
    cout << "I just got executed!" << endl;
}

Common Errors

  • It is a build error to call a function before its declaration.
  • It is a build error (linking error) to call a function when it is not defined.

Default Parameters

          
#include <iostream>
using namespace std;
void myFunction(string fname = "John", int age = 18);

    myFunction("Liam", 5);
    myFunction("Jenny");
    myFunction();
    return 0;
}

    cout << fname << " is " << age << " years old." << endl;
}

Return Values

          
#include <iostream>
using namespace std;
int myMax(int x, int y) {
    if (x < y)
        x = y;
    return x;
}
int main() {
    int x = 1;
    int y = 5;
    int z = myMax(x, y);
    cout << "The maximum value between " << x << " and " << y << " is " << z << endl;
    return 0;
}
