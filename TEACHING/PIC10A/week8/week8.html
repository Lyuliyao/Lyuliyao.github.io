Functions that take structs as parameters

          
            #include <iostream>
            struct Rectangle {
                double len1;
                double len2;
            };
            struct Square {
                double len;
            };
            struct Circle {
                double radius;
            };
            double area(Rectangle r) {
                return r.len1 * r.len2;
            }
            double area(Square s) {
                return s.len * s.len;
            }
            double area(Circle c) {
                return 3.14159 * c.radius * c.radius;
            }
            int main() {
                Rectangle r;
                r.len1 = 5;
                r.len2 = 10;

                double a = area(r); // a gets 50
                std::cout << "Area of rectangle: " << a << std::endl;

                Square s;
                s.len = 4;
                double a2 = area(s); // a2 gets 16
                std::cout << "Area of square: " << a2 << std::endl;

                Circle c;
                c.radius = 2.5;
                double a3 = area(c); // a3 gets 19.6349375
                std::cout << "Area of circle: " << a3 << std::endl;
                return 0;
            }

member functions

          
            #include <iostream>
            struct Rectangle {
                double len1;
                double len2;

                double area() const { // member function area
                    return len1 * len2;
                }
            };
            int main() {
                Rectangle r;
                r.len1 = 5;
                r.len2 = 10;

                double a = r.area(); // call member function area
                std::cout << "Area of rectangle: " << a << std::endl;
                return 0;
            }

multiple member functions

          
            #include <iostream>
            struct Rectangle {
                double len1;
                double len2;

                double area() const {
                    return len1 * len2;
                }

                double perimeter() const {
                    return 2 * (len1 + len2);
                }
            };
            int main() {
                Rectangle r;
                r.len1 = 5;
                r.len2 = 10;

                double a = r.area();
                double p = r.perimeter();
                std::cout << "Area of rectangle: " << a << std::endl;
                std::cout << "Perimeter of rectangle: " << p << std::endl;
                return 0;
            }

member functions with input

          
            #include <iostream>
            struct Rectangle {
                double len1;
                double len2;

                void set_dimensions(double l1, double l2) {
                    len1 = l1;
                    len2 = l2;
                }

                double area() const {
                    return len1 * len2;
                }
            };
            int main() {
                Rectangle r;
                r.set_dimensions(5, 10); // set dimensions using member function

                double a = r.area();
                std::cout << "Area of rectangle: " << a << std::endl;
                return 0;
            }

Struct Constructors

          
            #include <iostream>
            struct Rectangle {
                double len1;
                double len2;

                // Constructor
                Rectangle(double l1, double l2): len1(l1), len2(l2) {}

                double area() const {
                    return len1 * len2;
                }
            };
            int main() {
                Rectangle r(5, 10); // create Rectangle with len1=5, len2=10

                double a = r.area();
                std::cout << "Area of rectangle: " << a << std::endl;
                return 0;
            }

Struct Constructors with function body

          
            #include <iostream>
            using namespace std;
            struct Rectangle {
                double len1;
                double len2;

                // Constructor with function body
                Rectangle(double l1, double l2): len1(l1), len2(l2) {
                  if (len1 <= 0 || len2 <= 0) {
                    cout << "Warning: Lengths must be positive!" << std::endl;
                  }
                }

                double area() const {
                    return len1 * len2;
                }
            };
            int main() {
                Rectangle r(5, 10); // create Rectangle with len1=5, len2=10
                Rectangle r2(-3, 4); // Warning: Lengths must be positive!
                return 0;
            }
