Review of Structs

  • A structs (Structures) are a way to group related variables together.
  • Unlike vectors, structs can contain different data types.

Example of Structs 

        
          #include <iostream>
          using namespace std;
          struct C {
            int var;
          };
          int main() {
            C c;
            c.var = 5;
            cout << c.var << std::endl;
            return 0;
          }

member functions 

          
            #include <iostream>
            using namespace std;
            struct C{
              int var;
              void f() { ++var; cout << var <<  std::endl; }
            };
            int main() {
              C c;
              c.var = 5;
              c.f(); 
              return 0;
            }

const member functions 

          
            #include <iostream>
            using namespace std;
            struct C{
              int var;
              void f() { ++var; cout << var <<  std::endl; }
              void g() const { cout << var <<  std::endl; }
            };
            int main() {
              C c;
              c.var = 5;
              c.f(); 
              return 0;
            }

const objects

  • A const object is an object that cannot be modified after it is created.
          
            #include <iostream>
            using namespace std;
            struct C{
              int var;
              void f() { ++var; cout << var <<  std::endl; }
              void g() const { cout << var <<  std::endl; }
            };
            int main() {
              C c;
              c.var = 5;
              c.f(); 
              const C cc = c;
              cc.g();
              return 0;
            }

Summary of making member functions and instances const

  • The member variable of const instance cannot be modified after it is created.
  • The member variable of instance cannot be modified in a const member function.

Examples 

          
            #include <iostream>
            using namespace std;
            struct C{
              int var;
              void f() { ++var; cout << "no const: " << var <<  std::endl; }
              void g() const { cout << "const: " << var <<  std::endl; }
              void h() const;
            };
            int main() {
              C c;
              c.var = 5;
              c.f(); 
              c.g(); 
              const C cc = c;
              cc.f(); 
              cc.g();
              return 0;
            }
            void C::h() const {
              f();
              g();
              cout << "defined outside the struct" << std::endl;
            }

One more Example 

          
            #include <iostream>
            #include <cmath>
            using namespace std;
            
            struct Point {
              double x;
              double y;

              Point(double x_val, double y_val) : x(x_val), y(y_val) { }
              double distance(const Point& other) const {
                double dx = x - other.x;
                double dy = y - other.y;
                return sqrt(dx * dx + dy * dy);
              }
            };

            }

compute perimeter

          
            #include <iostream>
            #include <cmath>
            using namespace std;
            struct Point{
              double x;
              double y;
              Point() : x(0), y(0) { }
              Point(double x_val, double y_val) : x(x_val), y(y_val) { }
              double distance(const Point& other) const {
                double dx = x - other.x;
                double dy = y - other.y;
                return sqrt(dx * dx + dy * dy);
              }
            };
            struct Triangle{
              Point p1;
              Point p2;
              Point p3;
              Triangle(double x1, double y1,
                       double x2, double y2,
                       double x3, double y3)
                : p1(x1, y1), p2(x2, y2), p3(x3, y3) { }
              double perimeter() const {
                return p1.distance(p2) + p2.distance(p3) + p3.distance(p1);
              }
            };
            int main() {
              Triangle t(0,0,0,3,4,0); 
              cout << "Perimeter: " << t.perimeter() << std::endl;
              return 0;
            }

            #include <iostream>
            #include <cmath>
            using namespace std;
            struct Point{
              double x;
              double y;
              Point() : x(0), y(0) { }
              Point(double x_val, double y_val) : x(x_val), y(y_val) { }
              double distance(const Point& other) const {
                double dx = x - other.x;
                double dy = y - other.y;
                return sqrt(dx * dx + dy * dy);
              }
            };
            struct Triangle{
              Point p1;
              Point p2;
              Point p3;
              Triangle(double x1, double y1,
                       double x2, double y2,
                       double x3, double y3)
                : p1(x1, y1), p2(x2, y2), p3(x3, y3) { }
              Triangle(const Point& a, const Point& b, const Point& c)
                : p1(a), p2(b), p3(c) { }
              double perimeter() const {
                return p1.distance(p2) + p2.distance(p3) + p3.distance(p1);
              }
            };
            int main() {
              Point p1(0,0);
              Point p2(0,3);
              Point p3(4,0);
              Triangle t(p1, p2, p3);
              cout << "Perimeter: " << t.perimeter() << std::endl;
              return 0;
            }
